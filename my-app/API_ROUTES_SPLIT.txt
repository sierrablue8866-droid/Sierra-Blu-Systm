// app/api/leads/route.ts - Lead/Inquiry API Endpoint
import { NextRequest, NextResponse } from 'next/server';
import { db, addLead } from '@/lib/firebase-config';
import { collection, getDocs, query, where } from 'firebase/firestore';
import { propertyFinderService } from '@/lib/propertyFinder-service';

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { propertyCode, visitorName, visitorEmail, visitorPhone, message, source } = body;

    // Validation
    if (!propertyCode || !visitorName || !visitorEmail || !visitorPhone) {
      return NextResponse.json(
        { error: 'Missing required fields: propertyCode, visitorName, visitorEmail, visitorPhone' },
        { status: 400 }
      );
    }

    // Validate email format
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(visitorEmail)) {
      return NextResponse.json(
        { error: 'Invalid email format' },
        { status: 400 }
      );
    }

    // Add lead to Firestore
    const leadId = await addLead({
      propertyCode,
      visitorName,
      visitorEmail,
      visitorPhone,
      message: message || '',
      source: source || 'inquiry',
      status: 'new',
    });

    // Optional: Create lead in Property Finder too
    try {
      await propertyFinderService.createLead({
        propertyId: propertyCode,
        visitorName,
        visitorEmail,
        visitorPhone,
        message,
      });
      console.log('Lead synced to Property Finder');
    } catch (pfError) {
      console.warn('Property Finder sync failed (non-critical):', pfError);
      // Don't fail the request if Property Finder sync fails
    }

    return NextResponse.json(
      { 
        success: true, 
        leadId,
        message: 'Lead created successfully' 
      },
      { status: 201 }
    );
  } catch (error) {
    console.error('Lead creation error:', error);
    return NextResponse.json(
      { error: 'Failed to create lead', details: error instanceof Error ? error.message : 'Unknown error' },
      { status: 500 }
    );
  }
}

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const propertyCode = searchParams.get('propertyCode');
    const status = searchParams.get('status');

    const leadsRef = collection(db, 'leads');
    
    let q;
    const filters = [];

    if (propertyCode) {
      filters.push(where('propertyCode', '==', propertyCode));
    }
    if (status) {
      filters.push(where('status', '==', status));
    }

    if (filters.length > 0) {
      q = query(leadsRef, ...filters);
    } else {
      q = leadsRef;
    }

    const snapshot = await getDocs(q);
    const leads = snapshot.docs.map(doc => ({
      id: doc.id,
      ...doc.data(),
    }));

    return NextResponse.json({ 
      success: true,
      count: leads.length,
      leads 
    }, { status: 200 });
  } catch (error) {
    console.error('Get leads error:', error);
    return NextResponse.json(
      { error: 'Failed to fetch leads' },
      { status: 500 }
    );
  }
}

// app/api/viewing-requests/route.ts - Viewing Request API Endpoint
import { NextRequest, NextResponse } from 'next/server';
import { db, addViewingRequest } from '@/lib/firebase-config';
import { collection, getDocs, query, where } from 'firebase/firestore';

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const {
      propertyCode,
      visitorName,
      visitorEmail,
      visitorPhone,
      preferredDate,
      preferredTime,
      numberOfPeople,
      message,
    } = body;

    // Validation
    if (!propertyCode || !visitorName || !visitorEmail || !visitorPhone || !preferredDate) {
      return NextResponse.json(
        { error: 'Missing required fields: propertyCode, visitorName, visitorEmail, visitorPhone, preferredDate' },
        { status: 400 }
      );
    }

    // Validate email format
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(visitorEmail)) {
      return NextResponse.json(
        { error: 'Invalid email format' },
        { status: 400 }
      );
    }

    // Validate date is in future
    const requestDate = new Date(preferredDate);
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    
    if (requestDate < today) {
      return NextResponse.json(
        { error: 'Preferred date must be in the future' },
        { status: 400 }
      );
    }

    // Validate number of people
    const numPeople = numberOfPeople || 1;
    if (numPeople < 1) {
      return NextResponse.json(
        { error: 'Number of people must be at least 1' },
        { status: 400 }
      );
    }

    // Add viewing request to Firestore
    const requestId = await addViewingRequest({
      propertyCode,
      visitorName,
      visitorEmail,
      visitorPhone,
      preferredDate,
      preferredTime: preferredTime || '',
      numberOfPeople: numPeople,
      message: message || '',
      status: 'pending',
    });

    // TODO: Send confirmation email to visitor
    // Example:
    // await sendConfirmationEmail({
    //   to: visitorEmail,
    //   visitorName,
    //   propertyCode,
    //   preferredDate,
    //   preferredTime,
    // });

    console.log(`Viewing request created: ${requestId}`);

    return NextResponse.json(
      { 
        success: true, 
        requestId,
        message: 'Viewing request created successfully' 
      },
      { status: 201 }
    );
  } catch (error) {
    console.error('Viewing request creation error:', error);
    return NextResponse.json(
      { error: 'Failed to create viewing request', details: error instanceof Error ? error.message : 'Unknown error' },
      { status: 500 }
    );
  }
}

// app/api/properties/sync/route.ts - Property Sync API Endpoint
import { NextRequest, NextResponse } from 'next/server';
import { db } from '@/lib/firebase-config';
import { propertyFinderService } from '@/lib/propertyFinder-service';
import { collection, writeBatch, doc, Timestamp } from 'firebase/firestore';

export async function POST(request: NextRequest) {
  try {
    // Verify authorization
    const authHeader = request.headers.get('authorization');
    const expectedAuth = `Bearer ${process.env.SYNC_API_KEY}`;
    
    if (authHeader !== expectedAuth) {
      return NextResponse.json(
        { error: 'Unauthorized - Invalid API key' },
        { status: 401 }
      );
    }

    const body = await request.json();
    const { cityId } = body;

    if (!cityId) {
      return NextResponse.json(
        { error: 'Missing required field: cityId' },
        { status: 400 }
      );
    }

    console.log(`Starting sync for cityId: ${cityId}`);

    // Fetch properties from Property Finder
    const pfProperties = await propertyFinderService.syncPropertiesForCity(cityId);

    if (!pfProperties || pfProperties.length === 0) {
      return NextResponse.json(
        { 
          success: true,
          syncedCount: 0,
          message: 'No new properties found' 
        },
        { status: 200 }
      );
    }

    // Batch write to Firestore
    const batch = writeBatch(db);
    const listingsRef = collection(db, 'listings');

    let syncedCount = 0;
    const failedCount = 0;

    for (const property of pfProperties) {
      try {
        // Map Property Finder data to our schema
        const docRef = doc(listingsRef, property.id?.toString() || `pf_${Date.now()}_${syncedCount}`);
        
        batch.set(docRef, {
          code: property.id?.toString() || '',
          timestamp: Timestamp.fromDate(new Date(property.publishedDate || Date.now())),
          name: property.agent?.name || 'Unknown',
          mobile: property.agent?.phone || '',
          availability: property.isAvailable ? 'Available' : 'No answer',
          bedrooms: property.bedrooms || null,
          location: property.city?.name || 'Unknown',
          unitPrice: property.price || null,
          furnitureStatus: property.furnish?.name || null,
          propertyType: property.type?.name || 'Property',
          owner: property.postedBy === 'agent' ? 'Agent' : 'Owner',
          rentPeriodType: ['Yearly', 'Monthly'],
          agentName: property.agent?.name || 'Unknown',
          latitude: property.location?.latitude || null,
          longitude: property.location?.longitude || null,
          images: property.photos?.map((p: any) => p.url) || [],
          featured: false,
          createdAt: Timestamp.now(),
          updatedAt: Timestamp.now(),
        });
        
        syncedCount++;
      } catch (error) {
        console.error(`Error mapping property ${property.id}:`, error);
      }
    }

    // Commit batch in chunks (Firestore has a 500 write limit per batch)
    if (syncedCount > 0) {
      await batch.commit();
      console.log(`Successfully synced ${syncedCount} properties`);
    }

    return NextResponse.json(
      { 
        success: true, 
        syncedCount,
        failedCount,
        message: `Synced ${syncedCount} properties from Property Finder` 
      },
      { status: 200 }
    );
  } catch (error) {
    console.error('Property sync error:', error);
    return NextResponse.json(
      { 
        error: 'Failed to sync properties',
        details: error instanceof Error ? error.message : 'Unknown error'
      },
      { status: 500 }
    );
  }
}

export async function GET(request: NextRequest) {
  return NextResponse.json(
    { 
      message: 'Property Sync Endpoint',
      method: 'POST only',
      description: 'Use POST with Authorization header and cityId in body'
    },
    { status: 200 }
  );
}
